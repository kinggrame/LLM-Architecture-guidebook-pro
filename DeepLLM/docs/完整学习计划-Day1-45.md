## Day 3-4: nn.Module 与模型构建

### 2.1 定义神经网络

```python
import torch.nn as nn

class SimpleNet(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super().__init__()
        self.layer1 = nn.Linear(input_size, hidden_size)
        self.relu = nn.ReLU()
        self.layer2 = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        x = self.layer1(x)
        x = self.relu(x)
        x = self.layer2(x)
        return x

# 使用模型
model = SimpleNet(784, 128, 10)
output = model(input_tensor)

# Warmup + Cosine
def warmup_cosine(step, warmup_steps, total_steps):
    if step < warmup_steps:
        return step / warmup_steps
    else:
        progress = (step - warmup_steps) / (total_steps - warmup_steps)
        return 0.5 * (1 + np.cos(np.pi * progress))
```

# 第二阶段：Transformer 深入 (Day 11-20)

## Day 11-13: Multi-Head Attention 详解

### 6.1 注意力机制原理

```
Q (Query): "我在找什么" - 当前位置
K (Key): "我有什么" - 所有位置  
V (Value): "实际信息内容"

Attention(Q, K, V) = softmax(QK^T / √d) × V
```

### 6.2 逐行代码实现

```python
# src/models/transformer.py 中的 MultiHeadAttention 类

class MultiHeadAttention(nn.Module):
    def __init__(self, embed_dim, num_heads, dropout=0.1):
        super().__init__()
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        
        # 四个投影矩阵
        self.q_proj = nn.Linear(embed_dim, embed_dim)
        self.k_proj = nn.Linear(embed_dim, embed_dim)
        self.v_proj = nn.Linear(embed_dim, embed_dim)
        self.out_proj = nn.Linear(embed_dim, embed_dim)
        
        self.scale = math.sqrt(self.head_dim)
    
    def forward(self, q, k, v, mask=None):
        # 1. 投影
        Q = self.q_proj(q)
        K = self.k_proj(k)
        V = self.v_proj(v)
        
        # 2. 分头 (batch, seq, embed) -> (batch, heads, seq, head_dim)
        Q = Q.view(B, N, self.num_heads, self.head_dim).transpose(1, 2)
        K = K.view(B, N, self.num_heads, self.head_dim).transpose(1, 2)
        V = V.view(B, N, self.num_heads, self.head_dim).transpose(1, 2)
        
        # 3. 注意力分数
        scores = torch.matmul(Q, K.transpose(-2, -1)) / self.scale
        
        # 4. 掩码
        if mask is not None:
            scores = scores.masked_fill(mask == 0, float('-inf'))
        
        # 5. Softmax
        attn_weights = F.softmax(scores, dim=-1)
        
        # 6. 加权求和
        output = torch.matmul(attn_weights, V)
        
        # 7. 合并多头
        output = output.transpose(1, 2).contiguous().view(B, N, self.embed_dim)
        
        return self.out_proj(output), attn_weights
```

## Day 14-16: GPT 架构实现

### 7.1 GPT 与 BERT 的区别

| | GPT | BERT |
|---|---|---|
| 架构 | Decoder | Encoder |
| 注意力 | 单向 (Causal) | 双向 |
| 预训练 | 语言建模 (LM) | 掩码 MLM |
| 适用场景 | 生成 | 理解 |

### 7.2 GPT 模型代码

```python
# src/models/gpt.py 中的完整实现

class GPTBlock(nn.Module):
    """单层 GPT Transformer Block"""
    
    def __init__(self, embed_dim, num_heads, ff_dim):
        super().__init__()
        self.ln_1 = nn.LayerNorm(embed_dim)
        self.attn = GPTAttention(embed_dim, num_heads)
        self.ln_2 = nn.LayerNorm(embed_dim)
        self.mlp = GPTMLP(embed_dim, ff_dim)
    
    def forward(self, x, attention_mask=None):
        # Pre-LN 架构
        x = x + self.attn(self.ln_1(x), attention_mask)[0]
        x = x + self.mlp(self.ln_2(x))
        return x


class GPTModel(nn.Module):
    """完整 GPT 模型"""
    
    def __init__(self, vocab_size, embed_dim, num_layers, num_heads):
        super().__init__()
        
        # 词嵌入 + 位置嵌入
        self.wte = nn.Embedding(vocab_size, embed_dim)
        self.wpe = nn.Embedding(max_pos, embed_dim)
        
        # Transformer 层堆叠
        self.blocks = nn.ModuleList([
            GPTBlock(embed_dim, num_heads, ff_dim)
            for _ in range(num_layers)
        ])
        
        # 输出
        self.ln_f = nn.LayerNorm(embed_dim)
        self.lm_head = nn.Linear(embed_dim, vocab_size, bias=False)
        
        # 权重共享
        self.lm_head.weight = self.wte.weight
    
    def forward(self, input_ids, labels=None):
        # 嵌入
        x = self.wte(input_ids) + self.wpe(position_ids)
        
        # Transformer
        for block in self.blocks:
            x = block(x)
        
        # 输出
        logits = self.lm_head(self.ln_f(x))
        
        # 计算 loss
        loss = None
        if labels is not None:
            loss = F.cross_entropy(
                logits[:, :-1].contiguous().view(-1, vocab_size),
                labels[:, 1:].contiguous().view(-1)
            )
        
        return {'logits': logits, 'loss': loss}
```

### 7.3 SwiGLU 激活函数

```python
class GPTMLP(nn.Module):
    """使用 SwiGLU 的前馈网络"""
    
    def forward(self, x):
        # SwiGLU = SiLU(xW) * (xV) @ W
        return F.silu(self.fc1(x)) * self.fc2(x)) @ self.fc3(x)
```
